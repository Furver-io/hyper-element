Dataset Mutations

Demonstrates modifying dataset properties and their reflection to DOM attributes.

Adding Dataset Properties

Setting this.dataset.foo adds a data-foo attribute to the element.

customElements.define('dataset-add-element', class extends hyperElement {
  render(Html) {
    this.count = (this.count || 0) + 1;
    if (this.count > 1) return; // Prevent infinite re-render
    this.dataset.cats = 'dogs';
    Html`dataset-add-element`;
  }
});

document.getElementById('render-dataset-add').innerHTML = '<dataset-add-element></dataset-add-element>';

requestAnimationFrame(() => {
  const section = document.querySelector('[data-test="dataset-add"]');
  const elem = document.querySelector('dataset-add-element');
  section.dataset.testResult = (elem && elem.getAttribute('data-cats') === 'dogs') ? 'pass' : 'fail';
});

Updating Dataset with Array

Arrays assigned to dataset are serialized to JSON in the attribute.

customElements.define('dataset-update-element', class extends hyperElement {
  render(Html) {
    this.count = (this.count || 0) + 1;
    if (this.count > 1) return; // Prevent infinite re-render

    if (
      this.attrs['data-json'] !== '[5,6]' ||
      !Array.isArray(this.dataset.json) ||
      this.dataset.json.length !== 2
    ) {
      Html`dataset-update-element: initial check FAIL`;
      return;
    }
    this.dataset.json = [5, 6, 7];
    Html`dataset-update-element`;
  }
});

document.getElementById('render-dataset-update').innerHTML = '<dataset-update-element data-json="[5,6]"></dataset-update-element>';

requestAnimationFrame(() => {
  const section = document.querySelector('[data-test="dataset-update"]');
  const elem = document.querySelector('dataset-update-element');
  section.dataset.testResult = (elem && elem.getAttribute('data-json') === '[5,6,7]') ? 'pass' : 'fail';
});

CamelCase Dataset Properties

Dataset properties with camelCase map to kebab-case attributes.

customElements.define('dataset-update2-element', class extends hyperElement {
  render(Html) {
    this.count = (this.count || 0) + 1;
    if (this.count > 1) return; // Prevent infinite re-render

    if (
      this.attrs['data-a-b-c'] !== '[7,7,7]' ||
      typeof this.dataset.aBC !== 'object'
    ) {
      Html`dataset-update2-element: initial check FAIL`;
      return;
    }
    this.dataset.aBC = { zxy: 123 };
    Html`dataset-update2-element`;
  }
});

document.getElementById('render-dataset-update-camelcase').innerHTML = '<dataset-update2-element data-a-b-c="[7,7,7]"></dataset-update2-element>';

requestAnimationFrame(() => {
  const section = document.querySelector('[data-test="dataset-update-camelcase"]');
  const elem = document.querySelector('dataset-update2-element');
  section.dataset.testResult = (elem && elem.getAttribute('data-a-b-c') === '{"zxy":123}') ? 'pass' : 'fail';
});

Dynamically Adding Dataset Attribute

Adding a data attribute via setAttribute after element is connected should trigger attributeChangedCallback with oldVal=null. 

customElements.define('dataset-add-elem', class extends hyperElement {
  render(Html) {
    Html`Value: ${this.dataset.newKey || 'none'}`;
  }
});

document.getElementById('render-dataset-add-dynamic').innerHTML = '<dataset-add-elem></dataset-add-elem>';

requestAnimationFrame(() => {
  const section = document.querySelector('[data-test="dataset-add-dynamic"]');
  const elem = document.querySelector('dataset-add-elem');

  // Verify initial state - no attribute
  if (!elem || elem.dataset.newKey) {
    section.dataset.testResult = 'fail';
    console.error('Initial state wrong:', elem?.dataset?.newKey);
    return;
  }

  // Add the attribute dynamically - triggers attributeChangedCallback with oldVal=null
  elem.setAttribute('data-new-key', 'dynamic-value');

  // Wait for the callback and re-render
  setTimeout(() => {
    if (
      elem.textContent.includes('dynamic-value') &&
      elem.dataset.newKey === 'dynamic-value'
    ) {
      section.dataset.testResult = 'pass';
    } else {
      section.dataset.testResult = 'fail';
      console.error('Content did not update:', elem.textContent);
    }
  }, 100);
});

Removing CamelCase Dataset Properties

Removing a data attribute with hyphen (e.g., data-my-key) via removeAttribute should clean up this.dataset.myKey when observedAttributes includes it. 

customElements.define('dataset-remove-elem', class extends hyperElement {
  render(Html) {
    Html`Value: ${this.dataset.myKey || 'removed'}`;
  }
});

document.getElementById('render-dataset-remove-camelcase').innerHTML = '<dataset-remove-elem data-my-key="initial-value"></dataset-remove-elem>';

requestAnimationFrame(() => {
  const section = document.querySelector('[data-test="dataset-remove-camelcase"]');
  const elem = document.querySelector('dataset-remove-elem');

  // Verify initial state
  if (!elem || elem.dataset.myKey !== 'initial-value') {
    section.dataset.testResult = 'fail';
    console.error('Initial state wrong:', elem?.dataset?.myKey);
    return;
  }

  // Remove the attribute - triggers attributeChangedCallback
  elem.removeAttribute('data-my-key');

  // Wait for the callback and re-render
  setTimeout(() => {
    if (elem.textContent.includes('removed')) {
      section.dataset.testResult = 'pass';
    } else {
      section.dataset.testResult = 'fail';
      console.error('Content did not update:', elem.textContent);
    }
  }, 100);
});

Template Content Mutation (Observer Re-attach)

Tests that mutating innerHTML of a template element triggers re-parsing via observer.js attrs.template check.

Hello {name}! 
customElements.define(
  'template-mutation-elem',
  class extends hyperElement {
    render(Html) {
      Html`${Html.template({ name: 'World' })}`;
    }
  }
);

setTimeout(() => {
  const elem = document.querySelector('template-mutation-elem');
  const initialContent = elem.textContent;
  const hasInitial = initialContent.includes('Hello World!');

  // Mutate the template content - tests if observer.js line 60 is needed
  elem.innerHTML = 'Goodbye {name}!';

  setTimeout(() => {
    const finalContent = elem.textContent;
    // Note: Coverage shows observer.js line 60 (template re-attach) is NOT hit,
    // yet content changes. This suggests the line may be dead code OR
    // template mutation is handled elsewhere.
    const hasFinal = finalContent.includes('Goodbye World!');

    const section = document.querySelector('[data-test="template-content-mutation"]');
    if (hasInitial && hasFinal) {
      section.dataset.testResult = 'pass';
    } else {
      console.log('template-mutation: initial=' + initialContent + ', final=' + finalContent);
      section.dataset.testResult = 'fail';
    }
  }, 100);
}, 100);

Style Null Removes Attribute

Tests that style=${null} removes the style attribute entirely.

(async function () {
  const section = document.querySelector('[data-test="style-null-remove"]');
  const wait = () => new Promise(r => requestAnimationFrame(r));

  customElements.define(
    'style-null-elem',
    class extends hyperElement {
      setup(attachStore) {
        this.element._style = { color: 'red' };
        this.element._trigger = attachStore();
      }
      render(Html) {
        Html`<div id="style-null-result" style=${this.element._style}>styled</div>`;
      }
    }
  );

  const elem = document.createElement('style-null-elem');
  document.getElementById('style-null-output').appendChild(elem);

  await wait();
  const result = document.getElementById('style-null-result');
  const hasStyleBefore = result.hasAttribute('style');

  // Set style to null - should remove the attribute
  elem._style = null;
  elem._trigger();

  await wait();
  const hasStyleAfter = result.hasAttribute('style');
  const pass = hasStyleBefore === true && hasStyleAfter === false;
  section.dataset.testResult = pass ? 'pass' : 'fail';
  if (!pass) console.log('style-null: before=', hasStyleBefore, 'after=', hasStyleAfter);
})();

DIRECT: .property Setter

Tests the DIRECT path: return [path, direct(name.slice(1)), DIRECT] - triggered by .propName=${value} syntax.

(async function () {
  const section = document.querySelector('[data-test="direct-dot-prop"]');
  const wait = () => new Promise(r => requestAnimationFrame(r));

  customElements.define(
    'direct-dot-elem',
    class extends hyperElement {
      setup(onNext) {
        this.element._myValue = 42;
        this.element._checked = true;
        this.element._trigger = onNext();
      }
      render(Html) {
        // .value= and .checked= trigger DIRECT path with direct(name.slice(1))
        Html`<div id="direct-dot-result">
          <input type="text" id="direct-input" .value=${this.element._myValue} />
          <input type="checkbox" id="direct-checkbox" .checked=${this.element._checked} />
        </div>`;
      }
    }
  );

  const elem = document.createElement('direct-dot-elem');
  document.getElementById('direct-dot-output').appendChild(elem);

  await wait();
  const input = document.getElementById('direct-input');
  const checkbox = document.getElementById('direct-checkbox');

  // Verify direct property was set (not attribute)
  const inputValue = input.value;
  const inputAttr = input.getAttribute('value'); // Should be null (property, not attr)
  const checkboxChecked = checkbox.checked;

  // Update to verify reactivity
  elem._myValue = 100;
  elem._checked = false;
  elem._trigger();
  await wait();
  const updatedValue = input.value;
  const updatedChecked = checkbox.checked;

  const pass = inputValue === '42' &&
               inputAttr === null &&
               checkboxChecked === true &&
               updatedValue === '100' &&
               updatedChecked === false;
  section.dataset.testResult = pass ? 'pass' : 'fail';
  if (!pass) console.log('direct-dot:', { inputValue, inputAttr, checkboxChecked, updatedValue, updatedChecked });
})();

DIRECT: ref Attribute (Symbol Property)

Tests the ref path: return [path, directFor(ref), DIRECT] - stores callback on node via Symbol property.

(async function () {
  const section = document.querySelector('[data-test="ref-attribute"]');
  const wait = () => new Promise(r => requestAnimationFrame(r));

  // The callback we'll pass to ref - hyper-element stores it on the node
  const myRefCallback = (node) => { /* callback body */ };

  customElements.define(
    'ref-attr-elem',
    class extends hyperElement {
      setup(onNext) {
        this.element._refCallback = myRefCallback;
        this.element._trigger = onNext();
      }
      render(Html) {
        // ref=${callback} triggers directFor(ref) path
        // This stores the callback on the node via Symbol('ref')
        Html`<div id="ref-target" ref=${this.element._refCallback}>Ref Target</div>`;
      }
    }
  );

  const elem = document.createElement('ref-attr-elem');
  document.getElementById('ref-attr-output').appendChild(elem);

  await wait();

  const refTarget = document.getElementById('ref-target');

  // Verify: ref attribute is NOT rendered to HTML (it's a property, not an attribute)
  const noRefAttr = refTarget.getAttribute('ref') === null;

  // Verify: callback is stored on node via a Symbol property
  const symbols = Object.getOwnPropertySymbols(refTarget);
  const hasRefSymbol = symbols.some(s => s.description === 'ref');

  // Verify: the stored value IS our callback
  const refSymbol = symbols.find(s => s.description === 'ref');
  const storedCallback = refSymbol ? refTarget[refSymbol] : null;
  const callbackMatches = storedCallback === myRefCallback;

  const pass = noRefAttr && hasRefSymbol && callbackMatches;
  section.dataset.testResult = pass ? 'pass' : 'fail';
  if (!pass) console.log('ref-attr:', { noRefAttr, hasRefSymbol, callbackMatches, symbols: symbols.map(s => s.description) });
})();

Custom Overlay Hyper Layout Example

function defineWidget(tag, label) {
      if (customElements.get(tag)) return;
      hyperElement(tag, {
        render: (Html, ctx) => Html`<button class="card">${ctx.attrs.label || label}</button>`
      });
    }
    
    defineWidget('revenue-widget', 'Revenue');
    defineWidget('risk-widget', 'Risk');
    defineWidget('orders-widget', 'Orders');
    
    hyperElement('sample-overlay', {
      render: (Html, ctx) => {
        const can = ctx.attrs.can || [];
        const canDrag = can.includes('drag');
        const canResize = can.includes('resize');
        const itemId = ctx.attrs.item?.id || 'item';
    
        return Html`
          <div
            class="sample-overlay"
            style=${{ background: 'repeating-linear-gradient(135deg, #0f172a 0 12px, #1d4ed8 12px 24px)', border: '3px solid #f59e0b', boxShadow: 'inset 0 0 0 2px rgba(255, 255, 255, 0.45), 0 10px 26px rgba(15, 23, 42, 0.32)', boxSizing: 'border-box', color: '#ffffff', display: 'grid', height: '100%', overflow: 'hidden', placeItems: 'center', position: 'relative', width: '100%' }}
          >
          <button
            type="button"
            data-role="drag"
            aria-label="Move item"
            hidden=${!canDrag ? '' : null}
            onpointerdown=${(event) => ctx.attrs.drag(event)}
            style=${{ background: 'rgba(255, 255, 255, 0.94)', border: '2px solid #f59e0b', borderRadius: '999px', boxShadow: '0 8px 18px rgba(15, 23, 42, 0.24)', color: '#1e40af', cursor: 'grab', display: canDrag ? 'grid' : 'none', gap: '6px', minWidth: '138px', padding: '12px 16px', placeItems: 'center', position: 'relative' }}
          >
            <svg xmlns="http://www.w3.org/2000/svg" width="24" height="24" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round">
              <polyline points="5 9 2 12 5 15"></polyline>
              <polyline points="9 5 12 2 15 5"></polyline>
              <polyline points="15 19 12 22 9 19"></polyline>
              <polyline points="19 9 22 12 19 15"></polyline>
              <line x1="2" y1="12" x2="22" y2="12"></line>
              <line x1="12" y1="2" x2="12" y2="22"></line>
            </svg>
            <span style=${{ font: '700 13px/1.1 sans-serif', letterSpacing: '0', textTransform: 'uppercase' }}>
              ${itemId} editor
            </span>
          </button>
          <div
            aria-hidden="true"
            style=${{ background: '#f59e0b', bottom: '10px', color: '#111827', font: '700 11px/1 sans-serif', left: '10px', letterSpacing: '0', padding: '6px 8px', position: 'absolute' }}
          >
            ${itemId} overlay
          </div>
          <button
            type="button"
            data-role="resize"
            aria-label="Resize item"
            hidden=${!canResize ? '' : null}
            onpointerdown=${(event) => ctx.attrs.resize(event)}
            style=${{ background: '#f59e0b', border: '2px solid #ffffff', bottom: '8px', color: '#111827', cursor: 'nwse-resize', display: canResize ? 'grid' : 'none', height: '40px', padding: '6px', placeItems: 'center', position: 'absolute', right: '8px', width: '40px' }}
          >
            <svg xmlns="http://www.w3.org/2000/svg" width="24" height="24" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round">
              <line x1="7" y1="7" x2="17" y2="17"></line>
              <polyline points="17 7 17 17 7 17"></polyline>
            </svg>
          </button>
          </div>
        `;
      }
    });
    
    hyperElement('dashboard-with-overlay', {
      render: (Html) => {
        const items = [
          { id: 'revenue', tag: 'revenue-widget', can: ['drag'] },
          { id: 'risk', tag: 'risk-widget', can: ['drag', 'resize'] },
          { id: 'orders', tag: 'orders-widget', can: ['resize'] }
        ];
        const positions = {
          version: 1,
          columns: 12,
          items: [
            { id: 'risk', x: 0, y: 0, w: 4, h: 2 },
            { id: 'revenue', x: 4, y: 0, w: 4, h: 2 },
            { id: 'orders', x: 7, y: 0, w: 3, h: 2 }
          ]
        };
    
        return Html`
          <hyper-layout edit=${true} items=${items} positions=${positions} overlay="sample-overlay">
            <revenue-widget></revenue-widget>
            <risk-widget></risk-widget>
            <orders-widget></orders-widget>
          </hyper-layout>
        `;
      }
    });
    
    document.getElementById('custom-overlay-layout-output').innerHTML = '<dashboard-with-overlay></dashboard-with-overlay>';

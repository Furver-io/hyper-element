// Direct toString() exercise on the node classes from render/nodes.js.
// Each branch is asserted against a static string snapshot.
const m = await import('hyper-element/render/nodes.js').catch(() => null);
if (!m) {
  // bundle mode does not expose nodes.js — skip in that mode.
  window.__directToStringResult = 'skip';
} else {
  const { Element, Fragment, Text, Comment } = m;
  const errors = [];

  // Element with props and children
  const div = new Element('div', false);
  div.props = { id: 'test', class: 'container' };
  div.children = [new Text('Hello World')];
  if (div.toString() !== '<div id="test" class="container">Hello World</div>')
    errors.push('Element with children');

  // Element with boolean attribute (HTML mode)
  const button = new Element('button', false);
  button.props = { disabled: true };
  button.children = [new Text('Click')];
  const btnStr = button.toString();
  if (!btnStr.includes(' disabled') || !btnStr.includes('>Click</button>'))
    errors.push('Boolean attr HTML: ' + btnStr);

  // SVG element with boolean attribute (XML mode)
  const rect = new Element('rect', true);
  rect.props = { focusable: true, width: '100' };
  rect.children = [];
  const rectStr = rect.toString();
  if (!rectStr.includes('focusable=""') || !rectStr.includes(' />'))
    errors.push('Boolean attr XML: ' + rectStr);

  // Void element (no closing tag)
  const img = new Element('img', false);
  img.props = { src: 'test.png' };
  img.children = [];
  if (img.toString() !== '<img src="test.png">') errors.push('Void element');

  // Empty non-void element
  const span = new Element('span', false);
  span.props = {};
  span.children = [];
  if (span.toString() !== '<span></span>') errors.push('Empty non-void');

  // TEXT_ELEMENTS (style)
  const style = new Element('style', false);
  style.props = {};
  style.children = [new Text('.box { color: red; }')];
  if (!style.toString().includes('.box { color: red; }')) errors.push('TEXT_ELEMENT');

  // Fragment toString
  const frag = new Fragment();
  const c1 = new Element('span', false); c1.props = {}; c1.children = [new Text('A')];
  const c2 = new Element('span', false); c2.props = {}; c2.children = [new Text('B')];
  frag.children = [c1, c2];
  if (frag.toString() !== '<span>A</span><span>B</span>') errors.push('Fragment');

  // Comment toString
  const comment = new Comment('test comment');
  if (comment.toString() !== '<!--test comment-->') errors.push('Comment');

  // Text toString
  const textNode = new Text('plain text');
  if (textNode.toString() !== 'plain text') errors.push('Text');

  // Null attribute value (should be skipped)
  const divNull = new Element('div', false);
  divNull.props = { title: null, class: 'test' };
  divNull.children = [];
  const divNullStr = divNull.toString();
  if (divNullStr.includes('title') || !divNullStr.includes('class="test"'))
    errors.push('Null attr: ' + divNullStr);

  window.__directToStringResult = errors.length === 0 ? 'pass' : ('fail: ' + errors.join('; '));
}

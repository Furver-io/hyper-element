untracked() - Read Without Tracking

Tests that untracked reads don't create dependencies.

(function () {
      const section = document.querySelector('[data-test="untracked"]');
      try {
        const tracked = signal(1);
        const notTracked = signal(100);
        let effectRuns = 0;

        effect(() => {
          effectRuns++;
          const t = tracked.value; // This creates dependency
          const n = untracked(() => notTracked.value); // This doesn't
        });

        // Initial run
        const afterInit = effectRuns;

        // Change tracked - should trigger effect
        tracked.value = 2;
        const afterTracked = effectRuns;

        // Change not tracked - should NOT trigger effect
        notTracked.value = 200;
        const afterNotTracked = effectRuns;

        const pass =
          afterInit === 1 &&
          afterTracked === 2 &&
          afterNotTracked === 2;
        section.dataset.testResult = pass ? 'pass' : 'fail';
        document.getElementById('untracked-output').textContent =
          `init=${afterInit}, afterTracked=${afterTracked}, afterNotTracked=${afterNotTracked}`;
      } catch (e) {
        section.dataset.testResult = 'fail';
        document.getElementById('untracked-output').textContent = e.message;
      }
    })();

JSON Render Custom Components

Demonstrates using renderSpec programmatically alongside registerComponent to introduce highly customized schemas to the parser.

Programmatic Custom Component

Register a specific element type "CustomHeading" and embed it directly into hyper-element context.

registerComponent('CustomHeading', (Html, def, key) => {
  return Html.wire(def, ':' + key)`<h3 class="custom-heading">Hello ${def.props.name}</h3>`;
});
customElements.define('json-custom-wrapper', class extends hyperElement {
  render(Html) {
    return renderSpec(Html, {
      root: 'head_0',
      elements: { 'head_0': { type: 'CustomHeading', props: { name: 'World' } } }
    }, this);
  }
});
document.getElementById('custom-output').innerHTML = '<json-custom-wrapper></json-custom-wrapper>';

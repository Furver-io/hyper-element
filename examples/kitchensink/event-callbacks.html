Event Callbacks

Demonstrates using attachStore() without arguments for external event callbacks.

External Event Trigger (No Store)

Calling attachStore() without arguments returns a trigger function for re-rendering. 

const eventCallbacks = [];
function onExternalEvent(cb) {
  eventCallbacks.push(cb);
}
function emitExternalEvent(data) {
  eventCallbacks.forEach((cb) => cb(data));
}

customElements.define(
  'event-elem',
  class extends hyperElement {
    constructor() {
      super();
      this.count = 0;
    }

    setup(attachStore) {
      const triggerRender = attachStore();

      onExternalEvent((data) => {
        this.count = data;
        triggerRender();
      });
    }

    render(Html) {
      Html`<div id="event-result">Count: ${this.count}</div>`;
    }
  }
);

setTimeout(() => {
  emitExternalEvent(5);
  setTimeout(() => {
    emitExternalEvent(10);
    setTimeout(() => {
      const section = document.querySelector('[data-test="event-callback-no-store"]');
      const result = document.querySelector('#event-result');
      section.dataset.testResult =
        (result && result.textContent.includes('Count: 10')) ? 'pass' : 'fail';
    }, 100);
  }, 100);
}, 100);

External Event with Data Passed to Render

The trigger function can also pass data directly to render().

const dataEventCallbacks = [];
function onDataEvent(cb) {
  dataEventCallbacks.push(cb);
}
function emitDataEvent(data) {
  dataEventCallbacks.forEach((cb) => cb(data));
}

customElements.define(
  'event-data-elem',
  class extends hyperElement {
    setup(attachStore) {
      const triggerRender = attachStore();

      onDataEvent((data) => {
        triggerRender(data);
      });
    }

    render(Html, data) {
      Html`<div id="event-data-result">Data: ${data || 'none'}</div>`;
    }
  }
);

setTimeout(() => {
  emitDataEvent('PASS');
  setTimeout(() => {
    const section = document.querySelector('[data-test="event-callback-with-data"]');
    const result = document.querySelector('#event-data-result');
    section.dataset.testResult =
      (result && result.textContent.includes('Data: PASS')) ? 'pass' : 'fail';
  }, 100);
}, 200);

Event Handler Array with Options

Tests event handlers passed as arrays with options: @click=${[handler, { capture: true }]} 

let capturePhaseClicked = false;
let onceClicked = 0;

customElements.define(
  'event-array-elem',
  class extends hyperElement {
    setup(attachStore) {
      this.element._trigger = attachStore();
    }

    handleCapture(e) {
      capturePhaseClicked = true;
    }

    handleOnce(e) {
      onceClicked++;
    }

    render(Html) {
      Html`<div>
        <button id="capture-btn" @click=${[() => this.handleCapture(), { capture: true }]}>Capture</button>
        <button id="once-btn" @click=${[() => this.handleOnce(), { once: true }]}>Once</button>
      </div>`;
    }
  }
);

const arraySection = document.querySelector('[data-test="event-handler-array"]');
const arrayElem = document.createElement('event-array-elem');
document.getElementById('event-handler-array-output').appendChild(arrayElem);

setTimeout(() => {
  const captureBtn = document.getElementById('capture-btn');
  const onceBtn = document.getElementById('once-btn');

  captureBtn?.click();

  onceBtn?.click();
  onceBtn?.click();
  onceBtn?.click();

  setTimeout(() => {
    const pass = capturePhaseClicked && onceClicked === 1;
    arraySection.dataset.testResult = pass ? 'pass' : 'fail';
  }, 50);
}, 100);

Event Handler Array Replacement

Tests that event handler arrays can be replaced (removes old, adds new). 

let handlerAClicks = 0;
let handlerBClicks = 0;

customElements.define(
  'event-replace-elem',
  class extends hyperElement {
    setup(attachStore) {
      this.element._useHandlerA = true;
      this.element._trigger = attachStore();
    }

    handlerA() {
      handlerAClicks++;
    }

    handlerB() {
      handlerBClicks++;
    }

    render(Html) {
      const handler = this.element._useHandlerA
        ? [() => this.handlerA(), { capture: true }]
        : [() => this.handlerB(), { capture: false }];
      Html`<button id="replace-btn" @click=${handler}>Click</button>`;
    }
  }
);

const replaceSection = document.querySelector('[data-test="event-handler-array-replace"]');
const replaceElem = document.createElement('event-replace-elem');
document.getElementById('event-handler-array-replace-output').appendChild(replaceElem);

setTimeout(() => {
  const btn = document.getElementById('replace-btn');

  btn?.click();
  const afterA = handlerAClicks;

  replaceElem._useHandlerA = false;
  replaceElem._trigger();

  setTimeout(() => {
    btn?.click();

    setTimeout(() => {
      const pass = afterA === 1 && handlerAClicks === 1 && handlerBClicks === 1;
      replaceSection.dataset.testResult = pass ? 'pass' : 'fail';
    }, 50);
  }, 50);
}, 100);

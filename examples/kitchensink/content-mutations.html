Content Mutations

Demonstrates re-rendering when innerHTML is changed externally.

Content Change Re-render

When innerHTML is modified, the element re-renders with updated wrappedContent. 

customElements.define(
  'change-content',
  class extends hyperElement {
    render(Html) {
      Html`Content is: ${this.wrappedContent}`;
    }
  }
);

const elem = document.createElement('change-content');
elem.textContent = 'initial';
document.getElementById('content-mut-output').appendChild(elem);

// Change content after initial render
setTimeout(() => {
  elem.innerHTML = 'CHANGED';
  // Verify re-render after mutation
  setTimeout(() => {
    const section = document.querySelector('[data-test="change-content"]');
    section.dataset.testResult = elem.textContent.includes('CHANGED') ? 'pass' : 'fail';
  }, 100);
}, 50);

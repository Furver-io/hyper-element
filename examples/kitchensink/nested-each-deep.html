Nested Each Deep Coverage Tests

Tests for nested {+each {property}} syntax to cover parseEachBlocks.js lines 140-192.

1. Nested Each with Property Reference

Uses {+each {items}} syntax to iterate over a nested array property.

customElements.define(
  'nested-each-property',
  class extends hyperElement {
    render(Html) {
      const categories = [
        { name: 'Fruits', items: ['Apple', 'Banana'] },
        { name: 'Veggies', items: ['Carrot', 'Broccoli'] }
      ];
      Html`<div id="nested-each-result-1">{+each ${categories}}
        <div class="category">{name}: {+each {items}}<span class="item">{...}</span>{-each}</div>
      {-each}</div>`;
    }
  }
);

const elem = document.createElement('nested-each-property');
document.getElementById('output-1').appendChild(elem);

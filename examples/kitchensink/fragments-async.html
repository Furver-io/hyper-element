Async Fragments

Demonstrates fragments with promises and placeholders for async loading.

Fragment with Promise

Fragment methods return objects with promises that resolve to content. 

customElements.define('fragments-element', class extends hyperElement {
  AsyncData() {
    return {
      once: true,
      placeholder: "Loading...",
      text: Promise.resolve("Loaded!")
    }
  }
  render(Html) {
    Html`Data: ${{ AsyncData: true }}`
  }
});

const elem = document.querySelector('fragments-element');
const initialSnapshot = '<!--<>-->Data: Loading...<!--</>-->';
const finalSnapshot = '<!--<>-->Data: Loaded!<!--</>-->';
let passed = true;
if (elem.innerHTML !== initialSnapshot) {
  console.error('Phase 1 failed. Expected:', initialSnapshot, 'Got:', elem.innerHTML);
  passed = false;
}
setTimeout(() => {
  if (elem.innerHTML !== finalSnapshot) {
    console.error('Phase 2 failed. Expected:', finalSnapshot, 'Got:', elem.innerHTML);
    passed = false;
  }
  const section = document.querySelector('[data-test="fragment-async"]');
  section.dataset.testResult = passed ? 'pass' : 'fail';
}, 100);

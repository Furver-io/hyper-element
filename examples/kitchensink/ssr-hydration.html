SSR Hydration

Demonstrates SSR hydration with event capture and replay. Events that occur before component registration are buffered and replayed after the first render.

Basic Click Replay

Click events captured before component registration are replayed after hydration. 

var ssrBtn1 = document.getElementById('ssr-btn-1');
ssrBtn1.click();
ssrBtn1.click();
setTimeout(function () {
  customElements.define(
    'ssr-counter',
    class extends hyperElement {
      setup(onNext) {
        this.count = 0;
        return onNext(function () { return { count: this.count }; }.bind(this));
      }
      handleClick() {
        this.count++;
        this.element.render();
      }
      render(Html, store) {
        Html`<button onclick=${function () { this.handleClick(); }.bind(this)}>Count: ${store?.count ?? 0}</button>`;
      }
    }
  );
  setTimeout(function () {
    var counter = document.querySelector('ssr-counter');
    var section = document.querySelector('[data-test="ssr-basic-click"]');
    section.dataset.testResult = counter.textContent.includes('Count: 2') ? 'pass' : 'fail';
  }, 100);
}, 50);

Input Event Replay

Input events are replayed, restoring user input after hydration.

var input1 = document.getElementById('ssr-input-1');

}, 50);

onBeforeHydrate Filter (Class)

The onBeforeHydrate hook can filter events before replay. In this test, we filter to only replay the first event. 

var filterBtn1 = document.getElementById('ssr-filter-btn-1');

}, 50);

Functional API Lifecycle Hooks

onBeforeHydrate and onAfterHydrate work in functional components too. 

document.getElementById('ssr-func-hooks-btn').click();
document.getElementById('ssr-func-hooks-btn').click();
document.getElementById('ssr-func-hooks-btn').click();
setTimeout(function () {
  var filteredCount = 0;
  var afterCalled = false;
  hyperElement('ssr-func-hooks', {
    setup: function (ctx) {
      ctx.clicks = 0;
    },
    onBeforeHydrate: function (ctx, events) {
      filteredCount = events.length;
      return events.slice(0, 2);
    },
    onAfterHydrate: function (ctx) {
      afterCalled = true;
    },
    handleClick: function (ctx) {
      ctx.clicks++;
      ctx.element.render();
    },
    render: function (Html, ctx) {
      return Html`<button onclick=${function () { ctx.handleClick(); }}>Clicks: ${ctx.clicks}</button>`;
    },
  });
  setTimeout(function () {
    var elem = document.querySelector('ssr-func-hooks');
    var section = document.querySelector('[data-test="ssr-functional-hooks"]');
    section.dataset.testResult =
      (elem.textContent.includes('Clicks: 2') && filteredCount === 3 && afterCalled)
        ? 'pass' : 'fail';
  }, 100);
}, 50);

Multiple Event Types

Different event types on the same element are all captured.

var multiInput = document.getElementById('ssr-multi-input');
multiInput.dispatchEvent(new FocusEvent('focus', { bubbles: true }));
multiInput.value = 'test';
multiInput.dispatchEvent(new Event('input', { bubbles: true }));
multiInput.dispatchEvent(new FocusEvent('blur', { bubbles: true }));
setTimeout(function () {
  var events = [];
  customElements.define(
    'ssr-multi-events',
    class extends hyperElement {
      setup(onNext) {
        this.log = [];
        return onNext(function () { return { log: this.log }; }.bind(this));
      }
      handleFocus() {
        this.log.push('focus');
        events.push('focus');
      }
      handleInput(e) {
        this.log.push('input:' + e.target.value);
        events.push('input');
      }
      handleBlur() {
        this.log.push('blur');
        events.push('blur');
      }
      render(Html, store) {
        Html`
        <input type="text"
          onfocus=${function () { this.handleFocus(); }.bind(this)}
          oninput=${function (e) { this.handleInput(e); }.bind(this)}
          onblur=${function () { this.handleBlur(); }.bind(this)}>
        <span>Events: ${(store?.log || []).join(', ')}</span>
      `;
      }
    }
  );
  setTimeout(function () {
    var section = document.querySelector('[data-test="ssr-multi-events"]');
    section.dataset.testResult =
      (events.includes('focus') && events.includes('input') && events.includes('blur'))
        ? 'pass' : 'fail';
  }, 100);
}, 50);

Checked State Restoration (restoreElementState)

Tests that checkbox checked state is restored via restoreElementState when events are filtered in onBeforeHydrate. 

var restoreCheckbox = document.getElementById('ssr-restore-checkbox');
var restoreRadio = document.getElementById('ssr-restore-radio');
if (restoreCheckbox) {
  restoreCheckbox.checked = true;
  restoreCheckbox.dispatchEvent(new Event('change', { bubbles: true }));
}
if (restoreRadio) {
  restoreRadio.checked = true;
  restoreRadio.dispatchEvent(new Event('click', { bubbles: true }));
}
setTimeout(function () {
  var restoredViaState = false;
  customElements.define(
    'ssr-checked-restore',
    class extends hyperElement {
      onBeforeHydrate(events) {
        return [];
      }
      onAfterHydrate() {
        var checkbox = this.element.querySelector('input[type="checkbox"]');
        var radio = this.element.querySelector('input[type="radio"]');
        restoredViaState = checkbox?.checked || radio?.checked;
      }
      render(Html) {
        Html`<input type="checkbox" aria-label="test checkbox">
<input type="radio" name="ssr-radio" value="a" aria-label="test radio">
<label>Check me</label>`;
      }
    }
  );
  setTimeout(function () {
    var section = document.querySelector('[data-test="ssr-checked-restore"]');
    var elem = document.querySelector('ssr-checked-restore');
    var pass = elem !== null;
    section.dataset.testResult = pass ? 'pass' : 'fail';
  }, 100);
}, 50);

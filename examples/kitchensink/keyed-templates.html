Keyed Template Tests

Tests for Html.wire() keyed list rendering with dynamic updates.

Wire Keyed List Update and Re-render

Tests that updating items and calling render works correctly.

(function () {
  const section = document.querySelector('[data-test="wire-update-rerender"]');

  customElements.define(
    'wire-update-rerender-elem',
    class extends hyperElement {
      setup(attachStore) {
        this.element._items = [
          { id: 1, name: 'First' },
          { id: 2, name: 'Second' },
        ];
        this.element._trigger = attachStore();
      }
      render(Html) {
        Html`<ul id="wire-update-rerender-list">${this.element._items.map(
          (item) => Html.wire(item, ':item')`<li data-id=${item.id}>${item.name}</li>`
        )}</ul>`;
      }
    }
  );

  const elem = document.createElement('wire-update-rerender-elem');
  document.getElementById('wire-update-rerender-output').appendChild(elem);

  setTimeout(() => {
    const list = document.getElementById('wire-update-rerender-list');
    const initialLis = list?.querySelectorAll('li');
    const initialCount = initialLis?.length;

    elem._items = [
      { id: 1, name: 'Updated First' },
      { id: 2, name: 'Updated Second' },
      { id: 3, name: 'New Third' },
    ];
    elem._trigger();

    setTimeout(() => {
      const newLis = list?.querySelectorAll('li');
      const pass =
        initialCount === 2 &&
        newLis?.length === 3 &&
        newLis[0].textContent === 'Updated First' &&
        newLis[1].textContent === 'Updated Second' &&
        newLis[2].textContent === 'New Third';
      section.dataset.testResult = pass ? 'pass' : 'fail';
    }, 100);
  }, 100);
})();

Wire Keyed List Reorder

Tests reordering keyed items via render.

(function () {

  const itemA = { id: 'a', text: 'Alpha' };
  const itemB = { id: 'b', text: 'Beta' };

        this.element._items = [itemA, itemB, itemC];

Wire Object Identity

Tests that same object reference reuses wired content.

(function () {
  const section = document.querySelector('[data-test="wire-identity"]');

  const item = { id: 1, name: 'Stable' };

  customElements.define(
    'wire-identity-elem',
    class extends hyperElement {
      setup(attachStore) {
        this.element._item = item;
        this.element._trigger = attachStore();
      }
      render(Html) {
        Html`<div id="wire-identity-content">
          ${Html.wire(this.element._item, ':view')`<span class="item">${this.element._item.name}</span>`}
        </div>`;
      }
    }
  );

  const elem = document.createElement('wire-identity-elem');
  document.getElementById('wire-identity-output').appendChild(elem);

  setTimeout(() => {
    const content = document.getElementById('wire-identity-content');
    const span = content?.querySelector('.item');
    const initialRef = span;
    const initialText = span?.textContent;

    item.name = 'Changed';
    elem._trigger();

    setTimeout(() => {
      const newSpan = content?.querySelector('.item');
      const newText = newSpan?.textContent;

      const pass =
        initialText === 'Stable' &&
        newText === 'Changed' &&
        initialRef === newSpan;
      section.dataset.testResult = pass ? 'pass' : 'fail';
    }, 100);
  }, 100);
})();

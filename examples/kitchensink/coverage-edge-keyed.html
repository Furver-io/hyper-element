Keyed Array Rendering (key attribute)

Tests keyed list with key attribute to trigger Keyed.get/set coverage.

(function () {
  const section = document.querySelector('[data-test="keyed-attr-rerender"]');

  // Use objects so we can track DOM node identity
  const item1 = { id: 'a', text: 'First' };
  const item2 = { id: 'b', text: 'Second' };
  const item3 = { id: 'c', text: 'Third' };

  customElements.define(
    'keyed-attr-elem',
    class extends hyperElement {
      setup(attachStore) {
        this.element._items = [item1, item2, item3];
        this.element._trigger = attachStore();
      }
      render(Html) {
        // Use Html.wire() with key attribute for keyed rendering
        Html`<ul id="keyed-attr-list">${this.element._items.map(
          item => Html.wire(item, item.id)`<li key=${item.id} data-key=${item.id}>${item.text}</li>`
        )}</ul>`;
      }
    }
  );

  const elem = document.createElement('keyed-attr-elem');
  document.getElementById('keyed-attr-output').appendChild(elem);

  setTimeout(() => {
    const list = document.getElementById('keyed-attr-list');
    const initialLis = list?.querySelectorAll('li');
    const initialIds = Array.from(initialLis || []).map(li => li.dataset.key);

    // Get reference to middle DOM node before reorder
    const middleNodeBefore = initialLis?.[1];

    // Reorder items - this should trigger Keyed.get() to find existing holes
    elem._items = [item3, item2, item1]; // Reverse order
    elem._trigger();

    setTimeout(() => {
      const newLis = list?.querySelectorAll('li');
      const newIds = Array.from(newLis || []).map(li => li.dataset.key);
      const middleNodeAfter = newLis?.[1]; // Middle position should still be 'b'

      // Verify: order changed, but 'b' should be same DOM node (keyed reuse)
      const orderChanged = newIds.join(',') === 'c,b,a';
      const nodeReused = middleNodeBefore === middleNodeAfter;

      const pass = initialIds.join(',') === 'a,b,c' && orderChanged && nodeReused;
      section.dataset.testResult = pass ? 'pass' : 'fail';
    }, 100);
  }, 100);
})();

Diff: Tail Matching Fast Path

Tests tail-matching optimization: [A,B,C]→[X,B,C]. Hits lines 64-65 (tail match), 88 (remove A from map), 46 (insert X at beginning).

(function () {
  const section = document.querySelector('[data-test="diff-tail-match"]');

  // CRITICAL: Create actual DOM nodes that persist across renders
  // Raw DOM nodes go through diff.js directly (not holed() bypass)
  const nodeA = document.createElement('span');
  const nodeB = document.createElement('span');
  const nodeC = document.createElement('span');
  const nodeX = document.createElement('span');
  nodeA.dataset.id = 'a'; nodeA.textContent = 'A';
  nodeB.dataset.id = 'b'; nodeB.textContent = 'B';
  nodeC.dataset.id = 'c'; nodeC.textContent = 'C';
  nodeX.dataset.id = 'x'; nodeX.textContent = 'X';

  customElements.define(
    'diff-tail-match-elem',
    class extends hyperElement {
      setup(attachStore) {
        this.element._nodeList = [nodeA, nodeB, nodeC];
        this.element._trigger = attachStore();
      }
      render(Html) {
        // Pass raw DOM nodes - these go through diff.js
        Html`<div>${this.element._nodeList}</div>`;
      }
    }
  );

  const container = document.getElementById('diff-tail-match-output');
  container.innerHTML = '<diff-tail-match-elem></diff-tail-match-elem>';
  const elem = container.querySelector('diff-tail-match-elem');

  setTimeout(() => {
    const shadow = elem.shadowRoot || elem;
    const getOrder = () => Array.from(shadow.querySelectorAll('span')).map(s => s.dataset.id).join(',');

    const before = getOrder(); // 'a,b,c'

    // Mutation: replace first, keep tail → triggers tail matching + map removal
    // Trace: A≠X (no head), C=C (tail), B=B (tail), map removal for A, insert X
    elem._nodeList = [nodeX, nodeB, nodeC];
    elem._trigger();

    setTimeout(() => {
      const after = getOrder(); // 'x,b,c'
      const pass = before === 'a,b,c' && after === 'x,b,c';
      section.dataset.testResult = pass ? 'pass' : 'fail';
      if (!pass) console.log('diff-tail-match:', { before, after });
    }, 100);
  }, 100);
})();

Diff: Combined Scenario

Complex multi-step mutations to maximize diff branch coverage across all paths.

(function () {
  const section = document.querySelector('[data-test="diff-combined"]');

  // Create persistent DOM nodes
  const nodeA = document.createElement('span');
  const nodeB = document.createElement('span');
  const nodeC = document.createElement('span');
  const nodeD = document.createElement('span');
  const nodeE = document.createElement('span');
  const nodeX = document.createElement('span');
  const nodeY = document.createElement('span');
  const nodeZ = document.createElement('span');
  nodeA.dataset.id = 'a'; nodeA.textContent = 'A';
  nodeB.dataset.id = 'b'; nodeB.textContent = 'B';
  nodeC.dataset.id = 'c'; nodeC.textContent = 'C';
  nodeD.dataset.id = 'd'; nodeD.textContent = 'D';
  nodeE.dataset.id = 'e'; nodeE.textContent = 'E';
  nodeX.dataset.id = 'x'; nodeX.textContent = 'X';
  nodeY.dataset.id = 'y'; nodeY.textContent = 'Y';
  nodeZ.dataset.id = 'z'; nodeZ.textContent = 'Z';

  customElements.define(
    'diff-combined-elem',
    class extends hyperElement {
      setup(attachStore) {
        this.element._nodeList = [nodeA, nodeB, nodeC, nodeD, nodeE];
        this.element._trigger = attachStore();
      }
      render(Html) {
        Html`<div>${this.element._nodeList}</div>`;
      }
    }
  );

  const container = document.getElementById('diff-combined-output');
  container.innerHTML = '<diff-combined-elem></diff-combined-elem>';
  const elem = container.querySelector('diff-combined-elem');

  setTimeout(() => {
    const shadow = elem.shadowRoot || elem;
    const getOrder = () => Array.from(shadow.querySelectorAll('span')).map(s => s.dataset.id).join(',');

    const step0 = getOrder(); // 'a,b,c,d,e'

    // Step 1: Multiple insertions with tail match
    elem._nodeList = [nodeX, nodeY, nodeA, nodeB, nodeC, nodeD, nodeE];
    elem._trigger();

    setTimeout(() => {
      const step1 = getOrder(); // 'x,y,a,b,c,d,e'
      const step1ok = step1 === 'x,y,a,b,c,d,e';

      // Step 2: Reverse swap + removal
      elem._nodeList = [nodeE, nodeY, nodeA, nodeX];
      elem._trigger();

      setTimeout(() => {
        const step2 = getOrder(); // 'e,y,a,x'
        const step2ok = step2 === 'e,y,a,x';

        // Step 3: Clear all and repopulate with new order
        elem._nodeList = [nodeZ, nodeC, nodeA];
        elem._trigger();

        setTimeout(() => {
          const step3 = getOrder(); // 'z,c,a'
          const step3ok = step3 === 'z,c,a';

          const pass = step0 === 'a,b,c,d,e' && step1ok && step2ok && step3ok;
          section.dataset.testResult = pass ? 'pass' : 'fail';
          if (!pass) console.log('diff-combined:', { step0, step1, step2, step3 });
        }, 100);
      }, 100);
    }, 100);
  }, 100);
})();

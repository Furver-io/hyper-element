Event Handler Array Replacement

Tests that event handler arrays can be replaced (removes old, adds new). 

let handlerAClicks = 0;
let handlerBClicks = 0;

customElements.define(
  'event-replace-elem',
  class extends hyperElement {
    setup(attachStore) {
      this.element._useHandlerA = true;
      this.element._trigger = attachStore();
    }

    handlerA() {
      handlerAClicks++;
    }

    handlerB() {
      handlerBClicks++;
    }

    render(Html) {
      const handler = this.element._useHandlerA
        ? [() => this.handlerA(), { capture: true }]
        : [() => this.handlerB(), { capture: false }];
      Html`<button id="replace-btn" @click=${handler}>Click</button>`;
    }
  }
);

const replaceSection = document.querySelector('[data-test="event-handler-array-replace"]');
const replaceElem = document.createElement('event-replace-elem');
document.getElementById('event-handler-array-replace-output').appendChild(replaceElem);

setTimeout(() => {
  const btn = document.getElementById('replace-btn');

  btn?.click();
  const afterA = handlerAClicks;

  replaceElem._useHandlerA = false;
  replaceElem._trigger();

  setTimeout(() => {
    btn?.click();

    setTimeout(() => {
      const pass = afterA === 1 && handlerAClicks === 1 && handlerBClicks === 1;
      replaceSection.dataset.testResult = pass ? 'pass' : 'fail';
    }, 50);
  }, 50);
}, 100);

Observer Isolation

Verifies that descendant renders do not cause ancestor hyper-elements to re-render.

Descendant ChildList Mutations Stay Local

A child updating its own light DOM should not make the parent think its wrapped content changed. 

const childCount = signal(0);
let parentRenderCount = 0;

hyperElement('observer-childlist-child', {
  setup: (ctx, onNext) => {
    const render = onNext(() => ({ count: childCount.value }));
    const stop = effect(() => render());
    ctx.bump = () => { childCount.value = childCount.value + 1; };
    return stop;
  },
  render: (Html, ctx, state) => Html`
    <button onclick=${ctx.bump}>count=${state.count}</button>
    <span>child=${state.count}</span>
  `,
});

customElements.define(
  'observer-childlist-parent',
  class extends hyperElement {
    render(Html) {
      parentRenderCount++;
      this.dataset.parentRenders = String(parentRenderCount);
      Html`<observer-childlist-child></observer-childlist-child>`;
    }
  }
);

document.getElementById('observer-childlist-output').innerHTML =
  '<observer-childlist-parent></observer-childlist-parent>';

setTimeout(() => {
  const parent = document.querySelector('observer-childlist-parent');
  const button = parent.querySelector('button');
  const rendersBeforeClick = parent.dataset.parentRenders;
  button.click();

  setTimeout(() => {
    const childText = parent.querySelector('observer-childlist-child').textContent.replace(/\s+/g, ' ').trim();
    const section = document.querySelector('[data-test="descendant-childlist-does-not-rerender-parent"]');
    const pass =
      parent.dataset.parentRenders === rendersBeforeClick &&
      childText.includes('count=1') &&
      childText.includes('child=1');
    section.dataset.testResult = pass ? 'pass' : 'fail';
    if (!pass) {
      console.error(
        'Expected child update without parent rerender. before=' +
        rendersBeforeClick +
        ' after=' + parent.dataset.parentRenders +
        ' childText=' + childText
      );
    }
  }, 100);
}, 100);

Descendant Attribute Mutations Stay Local

A child toggling its own descendant attributes should not trip the ancestor observer either. 

const disabledSig = signal(false);
let attrParentRenderCount = 0;

hyperElement('observer-attr-child', {
  setup: (ctx, onNext) => {
    const render = onNext(() => ({ disabled: disabledSig.value }));
    const stop = effect(() => render());
    ctx.toggle = () => { disabledSig.value = !disabledSig.value; };
    return stop;
  },
  render: (Html, ctx, state) => Html`
    <button
      onclick=${ctx.toggle}
      disabled=${state.disabled ? '' : null}
      title=${state.disabled ? 'disabled' : 'enabled'}
    >toggle</button>
  `,
});

customElements.define(
  'observer-attr-parent',
  class extends hyperElement {
    render(Html) {
      attrParentRenderCount++;
      this.dataset.parentRenders = String(attrParentRenderCount);
      Html`<observer-attr-child></observer-attr-child>`;
    }
  }
);

document.getElementById('observer-attr-output').innerHTML =
  '<observer-attr-parent></observer-attr-parent>';

setTimeout(() => {
  const parent = document.querySelector('observer-attr-parent');
  const button = parent.querySelector('button');
  const rendersBeforeClick = parent.dataset.parentRenders;
  button.click();

  setTimeout(() => {
    const section = document.querySelector('[data-test="descendant-attributes-does-not-rerender-parent"]');
    const pass =
      parent.dataset.parentRenders === rendersBeforeClick &&
      button.disabled === true &&
      button.getAttribute('title') === 'disabled';
    section.dataset.testResult = pass ? 'pass' : 'fail';
    if (!pass) {
      console.error(
        'Expected descendant attribute update without parent rerender. before=' +
        rendersBeforeClick +
        ' after=' + parent.dataset.parentRenders +
        ' disabled=' + button.disabled +
        ' title=' + button.getAttribute('title')
      );
    }
  }, 100);
}, 100);

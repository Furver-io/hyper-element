Hyper Layout Core Dashboard Editing

function defineWidget(tag, label) {
  if (customElements.get(tag)) return;
  hyperElement(tag, {
    handleClick: (ctx) => {
      ctx.element.dataset.clicked = 'true';
    },
    render: (Html, ctx) => Html`<button class="card" onclick=${(event) => ctx.handleClick(event)}>${ctx.attrs.label || label}</button>`
  });
}

defineWidget('revenue-widget', 'Revenue');
defineWidget('risk-widget', 'Risk');
defineWidget('orders-widget', 'Orders');
defineWidget('dashboard-widget', 'Widget');

hyperElement('dashboard-view', {
  setup: (ctx, onNext) => {
    ctx.state = {
      editing: false,
      nextId: 4,
      widgets: [
        { id: 'revenue', tag: 'dashboard-widget', label: 'Revenue' },
        { id: 'risk', tag: 'dashboard-widget', label: 'Risk' },
        { id: 'orders', tag: 'dashboard-widget', label: 'Orders' }
      ],
      positions: {
      version: 1,
      columns: 12,
      items: [
        { id: 'revenue', x: 0, y: 0, w: 4, h: 2 },
        { id: 'risk', x: 4, y: 0, w: 4, h: 2 },
        { id: 'orders', x: 8, y: 0, w: 4, h: 2 }
      ]
      }
    };
    ctx.renderDashboard = onNext(() => ctx.state);
  },
  addWidget: (ctx) => {
    const nextId = ctx.state.nextId;
    ctx.state.nextId += 1;
    ctx.state.widgets = [
      ...ctx.state.widgets,
      { id: `added-${nextId}`, tag: 'dashboard-widget', label: `Added ${nextId}` }
    ];
    ctx.renderDashboard();
  },
  persistPositions: (ctx, _event, positions) => {
    ctx.state.positions = positions;
  },
  toggleEdit: (ctx) => {
    ctx.state.editing = !ctx.state.editing;
    ctx.renderDashboard();
  },
  removeWidgets: (ctx, _event, id, positions) => {
    if (!id) return;
    ctx.state.widgets = ctx.state.widgets.filter((widget) => widget.id !== id);
    ctx.state.positions = {
      ...positions,
      items: positions.items.filter((item) => item.id !== id)
    };
    ctx.renderDashboard();
  },
  render: (Html, ctx, state) => {
    const items = state.widgets.map((widget) => ({
      id: widget.id,
      tag: widget.tag,
      can: ['drag', 'resize']
    }));
    const breakpoints = [
      { width: 500, columns: 1 },
      { width: 900, columns: 12 }
    ];
    return Html`
      <div
        data-layout-toolbar
        style=${{ alignItems: 'center', display: 'grid', gap: '12px', gridTemplateColumns: '1fr auto 1fr', marginBottom: '8px' }}
      >
        <button
          data-layout-edit-toggle
          type="button"
          style=${{ alignItems: 'center', display: 'inline-flex', gap: '6px', justifySelf: 'start', padding: '4px 8px', fontSize: '12px' }}
          onclick=${ctx.toggleEdit}
        >
          <svg data-layout-lock-icon="closed" xmlns="http://www.w3.org/2000/svg" width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round" aria-hidden="true" style=${{ display: state.editing ? 'none' : 'block' }}>
            <rect x="3" y="11" width="18" height="11" rx="2" ry="2"></rect>
            <path d="M7 11V7a5 5 0 0 1 10 0v4"></path>
          </svg>
          <svg data-layout-lock-icon="open" xmlns="http://www.w3.org/2000/svg" width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round" aria-hidden="true" style=${{ display: state.editing ? 'block' : 'none' }}>
            <rect x="3" y="11" width="18" height="11" rx="2" ry="2"></rect>
            <path d="M7 11V7a5 5 0 0 1 9.9-1"></path>
          </svg>
          ${state.editing ? 'Stop editing layout' : 'Edit layout'}
        </button>
        <button
          data-layout-add
          type="button"
          style=${{ alignItems: 'center', display: 'inline-flex', gap: '6px', justifySelf: 'center', padding: '4px 8px', fontSize: '12px' }}
          onclick=${ctx.addWidget}
        >
          <svg data-layout-add-icon xmlns="http://www.w3.org/2000/svg" width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round" aria-hidden="true">
            <path d="M21 16V8a2 2 0 0 0-1-1.73l-7-4a2 2 0 0 0-2 0l-7 4A2 2 0 0 0 3 8v8a2 2 0 0 0 1 1.73l7 4a2 2 0 0 0 2 0l7-4A2 2 0 0 0 21 16z"></path>
            <polyline points="3.27 6.96 12 12.01 20.73 6.96"></polyline>
            <line x1="12" y1="22.08" x2="12" y2="12"></line>
          </svg>
          Add element
        </button>
        <div
          id="dashboard-trash"
          data-layout-trash
          style=${{ alignItems: 'center', background: '#fee2e2', border: '1px dashed #dc2626', borderRadius: '6px', boxSizing: 'border-box', color: '#991b1b', display: 'flex', fontSize: '12px', height: '32px', justifyContent: 'center', justifySelf: 'end', width: '120px' }}
        >Trash</div>
      </div>
      <hyper-layout
        edit=${state.editing}
        items=${items}
        positions=${state.positions}
        breakpoints=${breakpoints}
        cell-height="120"
        removable="trash"
        trash="#dashboard-trash"
        onchange=${ctx.persistPositions}
        onremoved=${ctx.removeWidgets}
      >
        ${state.widgets.map((widget) => Html.wire(widget, ':widget')`
          <dashboard-widget label=${widget.label}></dashboard-widget>
        `)}
      </hyper-layout>
    `;
  }
});

document.getElementById('layout-output').innerHTML = '<dashboard-view></dashboard-view>';

function defineWidget(tag, label) {
  if (customElements.get(tag)) return;
  hyperElement(tag, {
    render: (Html, ctx) => Html`<button class="card">${ctx.attrs.label || label}</button>`
  });
}

defineWidget('revenue-widget', 'Revenue');
defineWidget('risk-widget', 'Risk');
defineWidget('orders-widget', 'Orders');

let positions = {
  version: 1,
  columns: 12,
  items: [
    { id: 'revenue', x: 0, y: 0, w: 4, h: 2 },
    { id: 'risk', x: 4, y: 0, w: 4, h: 2 },
    { id: 'orders', x: 8, y: 0, w: 4, h: 2 }
  ]
};

function persistPositions(_event, nextPositions) {
  positions = nextPositions;
}

hyperElement('controlled-dashboard', {
  render: (Html, ctx) => {
    const items = [
      { id: 'revenue', tag: 'revenue-widget', can: ['drag'] },
      { id: 'risk', tag: 'risk-widget', can: ['drag', 'resize'] },
      { id: 'orders', tag: 'orders-widget', can: ['resize'] }
    ];

    return Html`
      <hyper-layout edit=${true} cell-height="120" items=${items} positions=${positions} onchange=${persistPositions}>
        <revenue-widget></revenue-widget>
        <risk-widget></risk-widget>
        <orders-widget></orders-widget>
      </hyper-layout>
    `;
  }
});

document.getElementById('controlled-layout-output').innerHTML = '<controlled-dashboard></controlled-dashboard>';

computed() - Basic Usage

Tests creating a computed value that depends on signals.

(function () {
      const section = document.querySelector('[data-test="computed-basic"]');
      try {
        const a = signal(2);
        const b = signal(3);
        const sum = computed(() => a.value + b.value);

        const initial = sum.value;

        a.value = 10;
        const afterA = sum.value;

        b.value = 20;
        const afterB = sum.value;

        // Peek without tracking
        const peeked = sum.peek();

        const pass =
          initial === 5 && afterA === 13 && afterB === 30 && peeked === 30;
        section.dataset.testResult = pass ? 'pass' : 'fail';
        document.getElementById('computed-basic-output').textContent =
          `initial=${initial}, afterA=${afterA}, afterB=${afterB}, peeked=${peeked}`;
      } catch (e) {
        section.dataset.testResult = 'fail';
        document.getElementById('computed-basic-output').textContent = e.message;
      }
    })();

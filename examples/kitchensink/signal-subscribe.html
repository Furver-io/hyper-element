signal() - Subscribe

Tests subscribing to signal changes.

(function () {
      const section = document.querySelector('[data-test="signal-subscribe"]');
      try {
        const count = signal(0);
        let callCount = 0;
        let lastValue = null;

        const unsubscribe = count.subscribe(() => {
          callCount++;
          lastValue = count.value;
        });

        count.value = 1;
        count.value = 2;
        count.value = 2; // Same value - should NOT notify (standard reactive behavior)

        // Unsubscribe and change - should not call
        unsubscribe();
        count.value = 3;

        const pass = callCount === 2 && lastValue === 2;
        section.dataset.testResult = pass ? 'pass' : 'fail';
        document.getElementById('signal-subscribe-output').textContent =
          `callCount=${callCount}, lastValue=${lastValue}`;
      } catch (e) {
        section.dataset.testResult = 'fail';
        document.getElementById('signal-subscribe-output').textContent = e.message;
      }
    })();

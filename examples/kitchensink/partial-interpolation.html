Partial Interpolation (Issue #21)

Demonstrates partial attribute interpolation like class="prefix-${value}".

Simple Partial Interpolation

Attribute values with static prefix and dynamic suffix.

customElements.define(
  'simple-partial',
  class extends hyperElement {
    render(Html) {
      const type = 'primary';
      Html`<div class="btn-${type}" id="simple-partial-target"></div>`;
    }
  }
);

document.getElementById('simple-partial-output').appendChild(
  document.createElement('simple-partial')
);

Mixed Partial and Full Interpolations

Partial interpolation with other dynamic attributes.

window.__mixedClicked = false;
customElements.define(
  'mixed-attrs',
  class extends hyperElement {
    render(Html) {
      const classVal = 'suffix';
      const titleVal = 'My Title';
      const clickHandler = () => { window.__mixedClicked = true; };
      const content = 'Content Here';
      Html`<div class="prefix-${classVal}" title="${titleVal}" onclick=${clickHandler} id="mixed-attrs-target">${content}</div>`;
    }
  }
);

document.getElementById('mixed-attrs-output').appendChild(
  document.createElement('mixed-attrs')
);

requestAnimationFrame(() => {
  const expectedMounted = '<div id="mixed-attrs-output"><mixed-attrs><div id="mixed-attrs-target" title="My Title" class="prefix-suffix">Content Here</div></mixed-attrs></div>';
  const container = document.getElementById('mixed-attrs-output');
  const mountedOk = container.outerHTML === expectedMounted;
  if (!mountedOk) {
    console.error('Phase Mounted mismatch (mixed-attrs)\nExpected: ' + expectedMounted + '\nActual:   ' + container.outerHTML);
  }
  document.getElementById('mixed-attrs-target').click();
  const clickOk = window.__mixedClicked === true;
  const section = document.querySelector('[data-test="mixed-attrs"]');
  section.dataset.testResult = (mountedOk && clickOk) ? 'pass' : 'fail';
});

Re-render with Updated Values

Verifies partial interpolation works across multiple renders.

window.__updatePartialState = { count: 0 };
window.__updatePartialCount = 0;

customElements.define(
  'update-partial',
  class extends hyperElement {
    render(Html) {
      window.__updatePartialCount++;
      const count = window.__updatePartialState.count;
      Html`<div class="counter-${count}" id="update-partial-target">${count}</div>`;
    }
  }
);

const updElem = document.createElement('update-partial');
document.getElementById('update-partial-output').appendChild(updElem);
window.__updatePartialElem = updElem;

const expectedInitial = '<div id="update-partial-output"><update-partial><div id="update-partial-target" class="counter-0">0</div></update-partial></div>';
const expectedUpdated = '<div id="update-partial-output"><update-partial><div id="update-partial-target" class="counter-5">5</div></update-partial></div>';

requestAnimationFrame(() => {
  const container = document.getElementById('update-partial-output');
  const initialOk = container.outerHTML === expectedInitial;
  if (!initialOk) {
    console.error('Phase Initial mismatch (update-partial)\nExpected: ' + expectedInitial + '\nActual:   ' + container.outerHTML);
  }
  window.__updatePartialState.count = 5;
  window.__updatePartialElem.render();
  requestAnimationFrame(() => {
    const updatedOk = container.outerHTML === expectedUpdated;
    if (!updatedOk) {
      console.error('Phase Updated mismatch (update-partial)\nExpected: ' + expectedUpdated + '\nActual:   ' + container.outerHTML);
    }
    const section = document.querySelector('[data-test="update-partial"]');
    section.dataset.testResult = (initialOk && updatedOk) ? 'pass' : 'fail';
  });
});

Empty Result Removes Attribute

When partial interpolation results in empty string, attribute is removed. 

customElements.define(
  'empty-result',
  class extends hyperElement {
    render(Html) {
      const state = { a: '', b: '' };
      // ${a}${b} with both empty = '' which removes attribute
      Html`<div class="${state.a}${state.b}" id="empty-result-target">Content</div>`;
    }
  }
);

document.getElementById('empty-result-output').appendChild(
  document.createElement('empty-result')
);

effect() - Cleanup Function

Tests that effect cleanup functions are called.

(function () {
      const section = document.querySelector('[data-test="effect-cleanup"]');
      try {
        const val = signal('a');
        let cleanupCount = 0;
        let effectCount = 0;

        const dispose = effect(() => {
          effectCount++;
          const v = val.value; // Track dependency
          return () => {
            cleanupCount++;
          };
        });

        // Initial run - no cleanup yet
        const cleanupAfterInit = cleanupCount;
        const effectAfterInit = effectCount;

        // Change - should run cleanup then new effect
        val.value = 'b';
        const cleanupAfterChange = cleanupCount;
        const effectAfterChange = effectCount;

        // Dispose - should run cleanup
        dispose();
        const cleanupAfterDispose = cleanupCount;

        const pass =
          cleanupAfterInit === 0 &&
          effectAfterInit === 1 &&
          cleanupAfterChange === 1 &&
          effectAfterChange === 2 &&
          cleanupAfterDispose === 2;
        section.dataset.testResult = pass ? 'pass' : 'fail';
        document.getElementById('effect-cleanup-output').textContent =
          `init: cleanup=${cleanupAfterInit} effect=${effectAfterInit}, change: cleanup=${cleanupAfterChange} effect=${effectAfterChange}, dispose: cleanup=${cleanupAfterDispose}`;
      } catch (e) {
        section.dataset.testResult = 'fail';
        document.getElementById('effect-cleanup-output').textContent = e.message;
      }
    })();

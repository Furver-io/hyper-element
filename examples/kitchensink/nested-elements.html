Nested Elements

Demonstrates parent-child custom element composition with attribute passing.

Passing String Attributes

Parent elements can pass string attributes to child custom elements. 

customElements.define(
  'inner1-elem',
  class extends hyperElement {
    render(Html) {
      Html`Inner: ${this.attrs.message || 'missing'}`;
    }
  }
);

customElements.define(
  'outter1-elem',
  class extends hyperElement {
    render(Html) {
      Html`Outer(<inner1-elem message=${'PASS'}></inner1-elem>)`;
    }
  }
);

document.getElementById('nested-string-attrs-output').appendChild(
  document.createElement('outter1-elem')
);

Passing Function References

Parent elements can pass function references to child elements.

customElements.define(
  'inner2-elem',
  class extends hyperElement {
    render(Html) {
      const fn = this.attrs.callback || (() => 'missing');
      Html`Result: ${fn()}`;
    }
  }
);

customElements.define(
  'outter2-elem',
  class extends hyperElement {
    getValue() { return 'PASS'; }
    render(Html) {
      Html`Outer(<inner2-elem callback=${this.getValue}/>)`;
    }
  }
);

document.getElementById('nested-function-attrs-output').appendChild(
  document.createElement('outter2-elem')
);

Different Element Type Composition

Parent elements can render child elements of a completely different semantic type, enabling realistic component composition. 

customElements.define(
  'status-badge',
  class extends hyperElement {
    render(Html) {
      const status = this.attrs.active ? 'online' : 'offline';
      Html`[${status}]`;
    }
  }
);

customElements.define(
  'user-card',
  class extends hyperElement {
    render(Html) {
      Html`${this.attrs.name}: <status-badge active=${false}></status-badge>`;
    }
  }
);

const card = document.createElement('user-card');
card.setAttribute('name', 'Alice');
document.getElementById('different-element-types-output').appendChild(card);

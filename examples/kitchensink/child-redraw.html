Tests that child elements re-render when parent passes new values.

Issue #11: Child Re-draws on Parent Update

When a parent element updates and passes new attribute values, the child should re-render. 

class ChildElem extends hyperElement {
  render(Html) {
    Html`<span id="child-display">Name: ${this.attrs.name}</span>`;
  }
}

class ParentElem extends hyperElement {
  setup(attachStore) {
    let counter = 0;
    const interval = setInterval(() => {
      counter++;
      attachStore(() => counter)();
    }, 200);

    return () => clearInterval(interval);
  }

  render(Html, state) {
    const counter = state || 0;
    Html`<child-elem name="${'value-' + counter}"></child-elem>`;
  }
}

customElements.define('child-elem', ChildElem);
customElements.define('parent-elem', ParentElem);

const parent = document.createElement('parent-elem');
document.getElementById('redraw-container').appendChild(parent);

const expectedPhase1 = '<parent-elem><child-elem name="value-0"><span id="child-display">Name: value-0</span></child-elem></parent-elem>';
const expectedPhaseFinal = '<parent-elem><child-elem name="value-3"><span id="child-display">Name: value-3</span></child-elem></parent-elem>';

let phase1Passed = false;
let p1Actual = '';

requestAnimationFrame(() => {
  p1Actual = parent.outerHTML;
  phase1Passed = p1Actual === expectedPhase1;
});

const observer = new MutationObserver(() => {
  if (parent.outerHTML === expectedPhaseFinal) {
    observer.disconnect();
    const section = document.querySelector('[data-test="child-redraw-on-parent-update"]');
    if (!phase1Passed) {
      console.error('Snapshot mismatch at Phase 1.\nExpected: ' + expectedPhase1 + '\nActual: ' + p1Actual);
      section.dataset.testResult = 'fail';
    } else {
      section.dataset.testResult = 'pass';
    }
  }
});

observer.observe(parent, { childList: true, subtree: true, attributes: true });

setTimeout(() => {
  const section = document.querySelector('[data-test="child-redraw-on-parent-update"]');
  if (section.dataset.testResult !== 'pass') {
    console.error('Timeout waiting for final snapshot.\nExpected: ' + expectedPhaseFinal + '\nActual: ' + parent.outerHTML);
    section.dataset.testResult = 'fail';
  }
}, 1500);

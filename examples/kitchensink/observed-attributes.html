Attribute Reactivity Tests

Tests for automatic attribute reactivity (no observedAttributes needed).

Basic Attribute Reactivity

Tests that attributes trigger re-render WITHOUT declaring observedAttributes. 

(function () {
  const section = document.querySelector('[data-test="basic-reactive"]');

  customElements.define(
    'basic-reactive-elem',
    class extends hyperElement {
      render(Html) {
        Html`<div id="basic-reactive-result">name=${this.attrs.name}</div>`;
      }
    }
  );

  const elem = document.createElement('basic-reactive-elem');
  elem.setAttribute('name', 'initial');
  document.getElementById('basic-reactive-output').appendChild(elem);

  requestAnimationFrame(() => {
    elem.setAttribute('name', 'updated');

    requestAnimationFrame(() => {
      const result = document.getElementById('basic-reactive-result');
      if (result?.textContent === 'name=updated') {
        section.dataset.testResult = 'pass';
      } else {
        section.dataset.testResult = 'fail';
      }
    });
  });
})();

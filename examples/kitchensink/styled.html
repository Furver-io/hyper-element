+styled Inline Styling System

Demonstrates the +styled template syntax for react-outline-inspired inline styling with base styles, prop flags, logic functions, and color palettes.

1. Base Styles

Simple base styles keyed by tag name.

hyperElement('styled-base', {
  styled: [{
    h2: { color: 'red', fontSize: '24px' },
    span: { fontWeight: 'bold' }
  }],
  render: (Html) => Html`
    <h2+styled>Title</h2>
    <span+styled>Content</span>
  `
});

const elem = document.createElement('styled-base');
document.getElementById('output-1').appendChild(elem);

requestAnimationFrame(() => {
  const section = document.querySelector('[data-test="base-styles"]');
  const h2 = elem.querySelector('h2');
  const span = elem.querySelector('span');
  const h2HasColor = h2 && h2.style.color === 'red';
  const h2HasFontSize = h2 && (h2.style.fontSize === '24px' || h2.style['font-size'] === '24px');
  const spanHasBold = span && span.style.fontWeight === 'bold';
  section.dataset.testResult = (h2HasColor && h2HasFontSize && spanHasBold) ? 'pass' : 'fail';
});

  const footer = elem.querySelector('footer');
  const articleHasBorder = article && article.style.borderRadius === '8px';
  const headerHasBg = header && (header.style.backgroundColor === 'rgb(240, 240, 240)' || header.style.backgroundColor === '#f0f0f0');
  const h1HasFontSize = h1 && h1.style.fontSize === '18px';
  const sectionHasPadding = sectionEl && sectionEl.style.padding === '15px';
  const pHasLineHeight = p && p.style.lineHeight === '1.5';
  const footerHasAlign = footer && footer.style.textAlign === 'right';
  section.dataset.testResult = (articleHasBorder && headerHasBg && h1HasFontSize && sectionHasPadding && pHasLineHeight && footerHasAlign) ? 'pass' : 'fail';
});

29. Styled with Non-Prop-Flag Attributes

Styled elements can have regular HTML attributes that aren't treated as prop flags.

window.__npfClickCount = 0;
const handler = () => { window.__npfClickCount++; };

hyperElement('styled-with-attrs', {
  styled: [{ div: { padding: '10px', color: 'green' } }],
  render: (Html) => Html`
    <div+styled
      class="my-class"
      id="my-id"
      style=${{ padding: '10px' }}
      role="button"
      slot="content"
      data-test="value"
      aria-label="Test label"
      @click=${handler}
      .title=${'Direct prop'}
      ?hidden=${false}>
      Content with attributes
    </div>
  `,
});

const elem = document.createElement('styled-with-attrs');
document.getElementById('output-29').appendChild(elem);

requestAnimationFrame(() => {
  const section = document.querySelector('[data-test="non-prop-flags"]');
  const div = elem.querySelector('div');
  const hasClass = div.classList.contains('my-class');
  const hasId = div.id === 'my-id';
  const hasDataTest = div.dataset.test === 'value';
  const hasAriaLabel = div.getAttribute('aria-label') === 'Test label';
  const hasRole = div.getAttribute('role') === 'button';
  const hasTitle = div.title === 'Direct prop';
  const hasStyle = div.style.padding === '10px' && div.style.color === 'green';
  div.click();
  const clickWorks = window.__npfClickCount === 1;
  section.dataset.testResult = (hasClass && hasId && hasDataTest && hasAriaLabel && hasRole && hasTitle && hasStyle && clickWorks) ? 'pass' : 'fail';
});

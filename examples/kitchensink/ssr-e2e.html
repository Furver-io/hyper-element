var imports1 = await import('./ssr-e2e-component.mjs');
var tagName = imports1.tagName;
var definition = imports1.definition;
var imports2 = await import('./ssr-e2e-output.js');
var ssrHtml = imports2.ssrHtml;

function setResult(phase, passed, message) {
  var section = document.querySelector('[data-test="phase-' + phase + '"]');
  section.dataset.testResult = passed ? 'pass' : 'fail';
  var statusEl = document.getElementById('phase-' + phase.split('-')[0] + '-status');
  if (statusEl) statusEl.textContent = message;
}

// Phase 1-2: verify SSR output
var ssrOutputEl = document.getElementById('ssr-html-output');
ssrOutputEl.textContent = ssrHtml;
if (ssrHtml.includes('<ssr-e2e-counter>') && ssrHtml.includes('Count: 0')) {
  setResult('1-2-ssr', true, '');
} else {
  setResult('1-2-ssr', false, '');
}

// Phase 3: mount
var mountPoint = document.getElementById('ssr-mount-point');
mountPoint.innerHTML = ssrHtml;
var componentEl = mountPoint.querySelector('ssr-e2e-counter');
var isNotDefined = customElements.get('ssr-e2e-counter') === undefined;
var tagExists = componentEl !== null;
var htmlMatches = mountPoint.innerHTML === ssrHtml;
setResult('3-mount', tagExists && isNotDefined && htmlMatches,
  'tag=' + tagExists + ', undef=' + isNotDefined + ', html=' + htmlMatches);

// Phase 4: pre-hydrate clicks
var btn = document.getElementById('ssr-e2e-btn');
if (btn) {
  btn.click();
  btn.click();
  setResult('4-pre-hydrate', true, 'Clicked button 2 times before hydration');
} else {
  setResult('4-pre-hydrate', false, 'Button not found in SSR output');
}

// Phase 5: define + replay
await new Promise(function (r) { setTimeout(r, 50); });
class SsrE2eCounter extends hyperElement {
  setup(onNext) { return definition.setup.call(this, onNext); }
  handleClick() { return definition.handleClick.call(this); }
  render(Html, store) { return definition.render.call(this, Html, store); }
}
customElements.define(tagName, SsrE2eCounter);
await new Promise(function (r) { setTimeout(r, 100); });
var counterEl = document.querySelector('ssr-e2e-counter');
var replayedContent = counterEl.textContent;
setResult('5-replay', replayedContent.includes('Count: 2'),
  'After replay: ' + replayedContent.trim());

// Phase 6: post-hydrate
var hydratedBtn = counterEl.querySelector('button');
if (hydratedBtn) {
  hydratedBtn.click();
  await new Promise(function (r) { setTimeout(r, 50); });
  var post = counterEl.textContent;
  setResult('6-post-hydrate', post.includes('Count: 3'),
    'After post-hydrate click: ' + post.trim());
} else {
  setResult('6-post-hydrate', false, 'Button not found after hydration');
}

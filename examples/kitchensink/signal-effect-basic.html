effect() - Basic Usage

Tests creating an effect that runs when dependencies change.

(function () {
      const section = document.querySelector('[data-test="effect-basic"]');
      try {
        const count = signal(0);
        let effectRuns = 0;
        let lastSeen = null;

        effect(() => {
          effectRuns++;
          lastSeen = count.value;
        });

        // Effect runs immediately
        const afterInit = effectRuns;

        count.value = 1;
        const afterChange1 = effectRuns;

        count.value = 5;
        const afterChange2 = effectRuns;

        const pass =
          afterInit === 1 &&
          afterChange1 === 2 &&
          afterChange2 === 3 &&
          lastSeen === 5;
        section.dataset.testResult = pass ? 'pass' : 'fail';
        document.getElementById('effect-basic-output').textContent =
          `init=${afterInit}, change1=${afterChange1}, change2=${afterChange2}, lastSeen=${lastSeen}`;
      } catch (e) {
        section.dataset.testResult = 'fail';
        document.getElementById('effect-basic-output').textContent = e.message;
      }
    })();

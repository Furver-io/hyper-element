Render Lifecycle Coverage Edges

Edge cases for base render, setup, attachStore, innerShadow, teardown, and prototype methods.

Base render() Method (No Override)

Tests that a hyperElement without an overridden render() still works. 

customElements.define(
  'base-render-elem',
  class extends hyperElement {
    // Intentionally no render() method - uses base class empty stub
  }
);

const baseRenderElem = document.createElement('base-render-elem');
document
  .getElementById('base-render-output')
  .appendChild(baseRenderElem);

requestAnimationFrame(() => {
  const section = document.querySelector('[data-test="base-render"]');
  // Element should exist and be connected, even with empty render
  if (baseRenderElem && baseRenderElem.isConnected) {
    section.dataset.testResult = 'pass';
  } else {
    section.dataset.testResult = 'fail';
  }
});

Disconnected Callback (Teardown)

Tests that teardown function is called when element is removed.

let teardownCalled = false;

customElements.define(
  'teardown-elem',
  class extends hyperElement {
    setup(attachStore) {
      attachStore({ count: 0 });
      // Return teardown function
      return () => {
        teardownCalled = true;
      };
    }
    render(Html, store) {
      Html`<span>Count: ${store ? store.count : 0}</span>`;
    }
  }
);

const container = document.getElementById('disconnected-output');
const elem6 = document.createElement('teardown-elem');
container.appendChild(elem6);

setTimeout(() => {
  // Remove element to trigger disconnectedCallback
  container.removeChild(elem6);

  setTimeout(() => {
    const section = document.querySelector(
      '[data-test="disconnected"]'
    );
    if (teardownCalled) {
      section.dataset.testResult = 'pass';
    } else {
      section.dataset.testResult = 'fail';
    }
  }, 50);
}, 50);

Base onBeforeHydrate (No Override)

Tests that base onBeforeHydrate returns events unchanged when not overridden.

(function () {
  // Click before component defined
  document.getElementById('base-hooks-btn').click();

  setTimeout(() => {
    let clickCount = 0;

    customElements.define(
      'ssr-base-hooks',
      class extends hyperElement {
        // No onBeforeHydrate override - uses base class that returns events

        handleClick() {
          clickCount++;
        }

        render(Html) {
          Html`<button onclick=${() => this.handleClick()}>Click</button>`;
        }
      }
    );

    setTimeout(() => {
      const section = document.querySelector('[data-test="base-before-hydrate"]');
      // Click should be replayed since base onBeforeHydrate returns events unchanged
      section.dataset.testResult = clickCount >= 1 ? 'pass' : 'fail';
    }, 100);
  }, 800);
})();

hyperElement Prototype Methods

Explicitly accesses hyperElement prototype methods for V8 coverage tracking.

(function () {
  const section = document.querySelector('[data-test="prototype-methods"]');

  // Access class constructor and prototype to help V8 coverage
  const HE = hyperElement;
  const proto = HE.prototype;

  // Check that all expected methods exist on prototype
  const hasConnected = typeof proto.connectedCallback === 'function';
  const hasDisconnected = typeof proto.disconnectedCallback === 'function';
  const hasAttrChanged = typeof proto.attributeChangedCallback === 'function';
  const hasSetup = typeof proto.setup === 'function';
  const hasRender = typeof proto.render === 'function';
  const hasBeforeHydrate = typeof proto.onBeforeHydrate === 'function';
  const hasAfterHydrate = typeof proto.onAfterHydrate === 'function';
  const hasAddDataset = typeof proto.addDataset === 'function';
  const hasGetDataset = typeof proto.getDataset === 'function';
  const hasAttachAttrs = typeof proto.attachAttrs === 'function';
  const hasInnerShadow = Object.getOwnPropertyDescriptor(proto, 'innerShadow')?.get !== undefined;

  // Call base class methods that are safe to call directly
  const testEvents = [{ type: 'test' }];
  const resultEvents = proto.onBeforeHydrate.call({}, testEvents);
  const beforeHydrateWorks = resultEvents === testEvents;

  // onAfterHydrate is a no-op
  proto.onAfterHydrate.call({});

  // setup is a no-op
  proto.setup.call({}, () => {});

  // render is a no-op
  proto.render.call({}, () => {});

  // attributeChangedCallback is a no-op
  proto.attributeChangedCallback.call({}, 'name', 'old', 'new');

  const pass = hasConnected && hasDisconnected && hasAttrChanged &&
               hasSetup && hasRender && hasBeforeHydrate && hasAfterHydrate &&
               hasAddDataset && hasGetDataset && hasAttachAttrs && hasInnerShadow &&
               beforeHydrateWorks;

  section.dataset.testResult = pass ? 'pass' : 'fail';
})();

disconnectedCallback and Teardown

Tests that disconnectedCallback triggers teardown function.

(function () {
  const section = document.querySelector(
    '[data-test="disconnected-callback"]'
  );
  let teardownCalled = false;

  customElements.define(
    'disconnect-elem',
    class extends hyperElement {
      setup(onNext) {
        // Return teardown function
        return onNext(() => {
          teardownCalled = true;
        });
      }

      render(Html) {
        Html`<div>Connected</div>`;
      }
    }
  );

  const container = document.getElementById('disconnect-output');
  container.innerHTML = '<disconnect-elem></disconnect-elem>';
  const elem = container.querySelector('disconnect-elem');

  setTimeout(() => {
    // Remove element to trigger disconnectedCallback
    elem.remove();

    // Check if teardown was called
    setTimeout(() => {
      section.dataset.testResult = teardownCalled ? 'pass' : 'fail';
    }, 50);
  }, 100);
})();

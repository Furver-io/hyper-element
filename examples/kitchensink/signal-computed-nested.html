computed() - Nested Computed

Tests computed values that depend on other computed values.

(function () {
      const section = document.querySelector('[data-test="computed-nested"]');
      try {
        const base = signal(2);
        const doubled = computed(() => base.value * 2);
        const quadrupled = computed(() => doubled.value * 2);

        const initial = quadrupled.value;

        base.value = 5;
        const updated = quadrupled.value;

        const pass = initial === 8 && updated === 20;
        section.dataset.testResult = pass ? 'pass' : 'fail';
        document.getElementById('computed-nested-output').textContent =
          `initial=${initial}, updated=${updated}`;
      } catch (e) {
        section.dataset.testResult = 'fail';
        document.getElementById('computed-nested-output').textContent = e.message;
      }
    })();

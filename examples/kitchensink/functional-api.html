Functional API

Demonstrates the functional API for defining custom elements without classes. All four signatures are covered.

1. Definition with Tag (auto-registers)

Full definition object with automatic registration.

hyperElement('func-counter', {
  render: (Html, ctx) => Html`Count: ${ctx.attrs.count || 0}`
});

document.getElementById('output-1').innerHTML = '<func-counter count="42"></func-counter>';

2. Shorthand with Tag (auto-registers)

Just a render function with automatic registration.

hyperElement('func-hello', (Html, ctx) =>
  Html`Hello, ${ctx.attrs.name || 'World'}!`
);

document.getElementById('output-2').innerHTML = '<func-hello name="Functional"></func-hello>';

3. Definition without Tag (manual registration)

Returns class for use with customElements.define().

const ManualElem = hyperElement({
  render: (Html, ctx) => Html`Manual: ${ctx.attrs.value}`
});
customElements.define('func-manual', ManualElem);

document.getElementById('output-3').innerHTML = '<func-manual value="works"></func-manual>';

4. Shorthand without Tag (manual registration)

Just a render function, returns class for manual registration.

const SimpleElem = hyperElement((Html, ctx) =>
  Html`Simple: ${ctx.attrs.msg}`
);
customElements.define('func-simple', SimpleElem);

document.getElementById('output-4').innerHTML = '<func-simple msg="shorthand"></func-simple>';

5. With Setup and Teardown

Lifecycle management with setup returning a teardown function.

hyperElement('func-timer', {
  setup: (ctx, onNext) => {
    let count = 0;
    const render = onNext(() => ({ count }));
    const interval = setInterval(() => { count++; render(); }, 100);
    return () => clearInterval(interval);
  },
  render: (Html, ctx, store) => Html`Timer: ${store?.count || 0}`
});

var elem5 = document.createElement('func-timer');
document.getElementById('output-5').appendChild(elem5);

var expectedInitial5 = '<func-timer><!--<>-->Timer: 0<!--</>--></func-timer>';
var expectedTick3 = '<func-timer><!--<>-->Timer: 3<!--</>--></func-timer>';
var initial5 = elem5.outerHTML;

setTimeout(function () {
  var initialOk = initial5 === expectedInitial5;
  var tickOk = elem5.outerHTML === expectedTick3;
  if (!initialOk) console.error('Phase Initial mismatch (setup-teardown)\nExpected: ' + expectedInitial5 + '\nActual:   ' + initial5);
  if (!tickOk) console.error('Phase Tick3 mismatch (setup-teardown)\nExpected: ' + expectedTick3 + '\nActual:   ' + elem5.outerHTML);
  var section = document.querySelector('[data-test="setup-teardown"]');
  section.dataset.testResult = (initialOk && tickOk) ? 'pass' : 'fail';
  elem5.remove();
}, 350);

6. With Methods and Events

Custom methods with context passed as first argument.

hyperElement('func-clicker', {
  setup: (ctx, onNext) => {
    let clicks = 0;
    const render = onNext(() => ({ clicks }));
    ctx.increment = () => {
      clicks++;
      render();
    };
  },
  handleClick: (ctx, event) => ctx.increment(),
  render: (Html, ctx, store) =>
    Html`<button onclick=${(e) => ctx.handleClick(e)}>Clicks: ${store?.clicks || 0}</button>`
});

var elem6 = document.createElement('func-clicker');
document.getElementById('output-6').appendChild(elem6);

var expectedInitial6 = '<func-clicker><button>Clicks: 0</button></func-clicker>';
var expectedAfter2 = '<func-clicker><button>Clicks: 2</button></func-clicker>';
var initial6 = elem6.outerHTML;

setTimeout(function () {
  var btn = elem6.querySelector('button');
  btn.click();
  btn.click();
  setTimeout(function () {
    var initialOk = initial6 === expectedInitial6;
    var afterOk = elem6.outerHTML === expectedAfter2;
    if (!initialOk) console.error('Phase Initial mismatch (methods-events)\nExpected: ' + expectedInitial6 + '\nActual:   ' + initial6);
    if (!afterOk) console.error('Phase After2 mismatch (methods-events)\nExpected: ' + expectedAfter2 + '\nActual:   ' + elem6.outerHTML);
    var section = document.querySelector('[data-test="methods-events"]');
    section.dataset.testResult = (initialOk && afterOk) ? 'pass' : 'fail';
  }, 50);
}, 50);

7. Class-based Backward Compatibility

Traditional class extension still works.

class ClassicElement extends hyperElement {
  render(Html) {
    Html`Classic: ${this.attrs.msg}`;
  }
}
customElements.define('func-classic', ClassicElement);

document.getElementById('output-7').innerHTML = '<func-classic msg="still works"></func-classic>';

8. Automatic Attribute Reactivity

All attributes are reactive - no observedAttributes needed!

hyperElement('func-reactive', {
  render: (Html, ctx) => Html`Value: ${ctx.attrs.value}`
});

var elem8 = document.createElement('func-reactive');
elem8.setAttribute('value', 'initial');
document.getElementById('output-8').appendChild(elem8);

var expectedInitial8 = '<func-reactive value="initial"><!--<>-->Value: initial<!--</>--></func-reactive>';
var expectedUpdated8 = '<func-reactive value="updated"><!--<>-->Value: updated<!--</>--></func-reactive>';
var initial8 = elem8.outerHTML;

requestAnimationFrame(function () {
  elem8.setAttribute('value', 'updated');
  setTimeout(function () {
    var initialOk = initial8 === expectedInitial8;
    var updatedOk = elem8.outerHTML === expectedUpdated8;
    if (!initialOk) console.error('Phase Initial mismatch (observed-attrs)\nExpected: ' + expectedInitial8 + '\nActual:   ' + initial8);
    if (!updatedOk) console.error('Phase Updated mismatch (observed-attrs)\nExpected: ' + expectedUpdated8 + '\nActual:   ' + elem8.outerHTML);
    var section = document.querySelector('[data-test="observed-attrs"]');
    section.dataset.testResult = (initialOk && updatedOk) ? 'pass' : 'fail';
  }, 0);
});

9. Setup Cleanup Verification

Verifies the teardown function returned from setup() is called on disconnect.

window.cleanupTestCalled = false;

hyperElement('func-cleanup-test', {
  setup: (ctx, onNext) => {
    onNext();
    return () => { window.cleanupTestCalled = true; };
  },
  render: (Html, ctx) => Html`Cleanup Test`
});

var elem9 = document.createElement('func-cleanup-test');
document.getElementById('output-9').appendChild(elem9);

var expectedMounted9 = '<func-cleanup-test>Cleanup Test</func-cleanup-test>';
var expectedRemovedContainer9 = '<div id="output-9"></div>';
var mounted9 = elem9.outerHTML;

requestAnimationFrame(function () {
  elem9.remove();
  requestAnimationFrame(function () {
    var container = document.getElementById('output-9');
    var mountedOk = mounted9 === expectedMounted9;
    var removedOk = container.outerHTML === expectedRemovedContainer9;
    var cleanupOk = window.cleanupTestCalled === true;
    if (!mountedOk) console.error('Phase Mounted mismatch (setup-cleanup-verify)\nExpected: ' + expectedMounted9 + '\nActual:   ' + mounted9);
    if (!removedOk) console.error('Phase Removed mismatch (setup-cleanup-verify)\nExpected: ' + expectedRemovedContainer9 + '\nActual:   ' + container.outerHTML);
    var section = document.querySelector('[data-test="setup-cleanup-verify"]');
    section.dataset.testResult = (mountedOk && removedOk && cleanupOk) ? 'pass' : 'fail';
  });
});

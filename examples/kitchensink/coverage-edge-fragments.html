Persistent Fragment Replace (Drop)

Tests toggling multi-node fragments to trigger drop() function.

(function () {
  const section = document.querySelector('[data-test="fragment-replace"]');

  customElements.define(
    'fragment-replace-elem',
    class extends hyperElement {
      setup(attachStore) {
        this.element._showFragment = true;
        this.element._trigger = attachStore();
      }
      render(Html) {
        const showFrag = this.element._showFragment;
        if (showFrag) {
          Html`<div id="fragment-replace-result"><span class="frag-a">A</span><span class="frag-b">B</span></div>`;
        } else {
          Html`<div id="fragment-replace-result"><span class="frag-single">Single</span></div>`;
        }
      }
    }
  );

  const elem = document.createElement('fragment-replace-elem');
  document.getElementById('fragment-replace-output').appendChild(elem);

  setTimeout(() => {
    const result = document.getElementById('fragment-replace-result');
    const initialA = result?.querySelector('.frag-a');
    const initialB = result?.querySelector('.frag-b');

    // Toggle to single element (triggers drop to replace fragment)
    elem._showFragment = false;
    elem._trigger();

    setTimeout(() => {
      const newResult = document.getElementById('fragment-replace-result');
      const single = newResult?.querySelector('.frag-single');
      const noA = !newResult?.querySelector('.frag-a');
      const noB = !newResult?.querySelector('.frag-b');

      const pass = initialA && initialB && single && noA && noB;
      section.dataset.testResult = pass ? 'pass' : 'fail';
    }, 50);
  }, 100);
})();

(function () {
  const section = document.querySelector('[data-test="persistent-fragment-rerender"]');

  customElements.define(
    'persistent-frag-elem',
    class extends hyperElement {
      setup(attachStore) {
        this.element._count = 1;
        this.element._trigger = attachStore();
      }
      render(Html) {
        // Template with multiple root nodes triggers PersistentFragment
        Html`<span>A${this.element._count}</span><span>B${this.element._count}</span><span>C${this.element._count}</span>`;
      }
    }
  );

  const container = document.getElementById('persistent-fragment-output');
  container.innerHTML = '<persistent-frag-elem></persistent-frag-elem>';
  const elem = container.querySelector('persistent-frag-elem');

  setTimeout(() => {
    const shadow = elem.shadowRoot || elem;
    let spans = shadow.querySelectorAll('span');
    const initial = spans.length === 3;

    // Trigger re-render which exercises valueOf path
    elem._count = 2;
    elem._trigger();

    setTimeout(() => {
      spans = shadow.querySelectorAll('span');
      const afterRerender = spans.length === 3;
      const content = Array.from(spans).map(s => s.textContent).join(',');
      const pass = initial && afterRerender && content === 'A2,B2,C2';
      section.dataset.testResult = pass ? 'pass' : 'fail';
    }, 100);
  }, 100);
})();

PersistentFragment: valueOf() Else Branch

Tests that valueOf() correctly collects nodes from DOM when parentNode !== this. Direct test: create PersistentFragment, append to DOM (children move out), then call valueOf() which must collect children back. 

const { PersistentFragment, nodes } = await import('../../src/render/persistent-fragment.js');

const section = document.querySelector('[data-test="persistent-fragment-valueof-else"]');
const container = document.getElementById('pf-valueof-else-output');

try {
  // Create a DocumentFragment with multiple children
  const docFrag = document.createDocumentFragment();
  const span1 = document.createElement('span');
  span1.textContent = 'A';
  span1.className = 'pf-a';
  const span2 = document.createElement('span');
  span2.textContent = 'B';
  span2.className = 'pf-b';
  const span3 = document.createElement('span');
  span3.textContent = 'C';
  span3.className = 'pf-c';
  docFrag.appendChild(span1);
  docFrag.appendChild(span2);
  docFrag.appendChild(span3);

  // Wrap in PersistentFragment (adds comment markers)
  const pf = PersistentFragment(docFrag);

  // Verify initial state: children are in the fragment
  // parentNode getter returns this.firstChild.parentNode
  // At this point, firstChild (the start comment) is in the fragment
  const initialParent = pf.parentNode;
  const initialInFragment = initialParent === pf;
  console.log('pf-valueof-else: initial parentNode === pf:', initialInFragment);

  // Create a wrapper div to append the fragment to
  const wrapper = document.createElement('div');
  wrapper.className = 'pf-wrapper';
  container.appendChild(wrapper);

  // Append the PersistentFragment to DOM
  // This moves all children (comment markers + spans) into the wrapper
  wrapper.appendChild(pf);

  // After appending, the fragment is empty but its firstChild/lastChild
  // properties still reference the comment markers (now in DOM)
  const afterAppendParent = pf.parentNode;
  const afterAppendInDOM = afterAppendParent === wrapper;
  console.log('pf-valueof-else: after append parentNode === wrapper:', afterAppendInDOM);

  // Verify children are visible in DOM
  const visibleBefore = wrapper.querySelectorAll('span').length;
  console.log('pf-valueof-else: spans in DOM:', visibleBefore);

  // NOW: Call valueOf() - this should hit the ELSE branch!
  // parentNode !== this (it's the wrapper), so valueOf() must:
  // 1. Collect all nodes from firstChild to lastChild via nextSibling
  // 2. Call replaceChildren() to move them back into the fragment
  const result = pf.valueOf();

  // After valueOf(), the children should be back in the fragment
  const afterValueOfParent = pf.parentNode;
  const afterValueOfInFragment = afterValueOfParent === pf;
  console.log('pf-valueof-else: after valueOf parentNode === pf:', afterValueOfInFragment);

  // Verify the fragment contains the children
  const fragChildren = pf.childNodes.length;
  console.log('pf-valueof-else: fragment childNodes:', fragChildren);

  // Verify the nodes array was populated
  const nodesArray = pf[nodes];
  const nodesLength = nodesArray ? nodesArray.length : 0;
  console.log('pf-valueof-else: nodes array length:', nodesLength);

  // The wrapper should now be empty (children moved back to fragment)
  const visibleAfter = wrapper.querySelectorAll('span').length;
  console.log('pf-valueof-else: spans still in wrapper:', visibleAfter);

  // Verify result is the fragment itself
  const resultIsFragment = result === pf;

  // Success criteria:
  // 1. Initially children were in fragment (parentNode === pf)
  // 2. After append, children were in DOM (parentNode === wrapper)
  // 3. After valueOf(), children are back in fragment
  // 4. The nodes array was populated (5 nodes: comment, span, span, span, comment)
  const pass =
    initialInFragment &&
    afterAppendInDOM &&
    afterValueOfInFragment &&
    nodesLength === 5 && // 2 comments + 3 spans
    fragChildren === 5 &&
    visibleAfter === 0 && // children moved out of wrapper
    resultIsFragment;

  section.dataset.testResult = pass ? 'pass' : 'fail';
  console.log('pf-valueof-else result:', pass ? 'PASS' : 'FAIL');

  // Clean up by re-appending fragment to show content
  wrapper.appendChild(pf);

} catch (err) {
  console.error('pf-valueof-else error:', err);
  section.dataset.testResult = 'fail';
}

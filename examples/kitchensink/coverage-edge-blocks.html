Blocks Coverage Edges

Edge cases for {+if}, {+each}, {+unless} block syntax: empty arrays, undefined arrays, with-index, else branches, truthy/falsy, template-form blocks, and multiroot template arrays.

Each with Non-Array Value

Tests that {#each} with non-array returns empty string.

(function () {
  const section = document.querySelector(
    '[data-test="each-non-array"]'
  );

  customElements.define(
    'each-non-array-elem',
    class extends hyperElement {
      List(data) {
        return {
          template: '{+each items}<li>{...}</li>{-each}',
          values: data,
        };
      }
      render(Html) {
        // Pass non-array for items
        Html`<ul>${{ List: { items: 'not-an-array' } }}</ul>`;
      }
    }
  );

  const elem = document.createElement('each-non-array-elem');
  document.getElementById('each-non-array-output').appendChild(elem);

  setTimeout(() => {
    // Should render empty list (no li elements)
    const liCount = elem.querySelectorAll('li').length;
    section.dataset.testResult = liCount === 0 ? 'pass' : 'fail';
  }, 100);
})();

Template Literal {+if} Block

Tests {+if} block using template literal syntax (transformEachBlocks path).

(function () {
  const section = document.querySelector('[data-test="tpl-if-block"]');

  customElements.define(
    'tpl-if-elem',
    class extends hyperElement {
      setup(onNext) {
        this.showA = true;
        this.showB = false;
        onNext();
      }
      render(Html) {
        Html`<div id="tpl-if-result">[{+if ${this.showA}}TRUTHY{-if}][{+if ${this.showB}}HIDDEN{-if}]</div>`;
      }
    }
  );

  const elem = document.createElement('tpl-if-elem');
  document.getElementById('tpl-if-output').appendChild(elem);

  requestAnimationFrame(() => {
    const result = document.getElementById('tpl-if-result');
    const pass = result?.textContent === '[TRUTHY][]';
    section.dataset.testResult = pass ? 'pass' : 'fail';
  });
})();

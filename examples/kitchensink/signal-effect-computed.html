effect() with computed()

Tests effects that depend on computed values.

(function () {
      const section = document.querySelector('[data-test="effect-computed"]');
      try {
        const count = signal(1);
        const doubled = computed(() => count.value * 2);
        let effectValue = null;
        let effectRuns = 0;

        effect(() => {
          effectRuns++;
          effectValue = doubled.value;
        });

        const afterInit = effectRuns;
        const valueInit = effectValue;

        count.value = 5;
        const afterChange = effectRuns;
        const valueChange = effectValue;

        const pass =
          afterInit === 1 &&
          valueInit === 2 &&
          afterChange === 2 &&
          valueChange === 10;
        section.dataset.testResult = pass ? 'pass' : 'fail';
        document.getElementById('effect-computed-output').textContent =
          `init: runs=${afterInit} value=${valueInit}, change: runs=${afterChange} value=${valueChange}`;
      } catch (e) {
        section.dataset.testResult = 'fail';
        document.getElementById('effect-computed-output').textContent = e.message;
      }
    })();
